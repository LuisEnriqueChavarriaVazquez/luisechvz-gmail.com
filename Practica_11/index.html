// Estructura de datos para STACK

                

struct stack{
    int superior;		
    int maximoTamaño;
    int *elementos;
};
                

                            
                
// Funcion para inicializar nuestra PILA

                

struct stack* newStack(int capacidad){
    struct stack *nodo = (struct stack*)malloc(sizeof(struct stack));
    
    nodo->maximoTamaño = capacidad;
    nodo->superior = -1;
    nodo->elementos = (int*)malloc(sizeof(int) * capacidad);
    
    return nodo;
}
                


                
// Utility function to add an element x in the stack

                

void push(struct stack *nodo, int x){
    //En caso de OVERFLOW
    if (isFull(nodo)){

}
